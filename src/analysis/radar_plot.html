Remove the columns for results we do not wish to plot in radar plot.

# Remove results without offset correction on new site (+ new preprocessing) dataset
df2 <- df[,-seq(7,16)]
# Remove all standard deviations MEsd, MAEsd etc. and ME for new site datasets with correction
df2 <- df2[,-c(3,5, #MAsd, MAEsd
               7,8,10, # MA, MAsd, MAEsd -- new site same preproc
               12, 13,15, # MA, MAsd, MAEsd -- new site diff preproc
               23,25)] # MdEsd, MAdEsd

# take absolute value of ME and MdE -- here, we care how far from 0 ME is, but not the direction of the error
df2$multisite.ME <- abs(df2$multisite.ME)
df2$MdE <- abs(df2$MdE)
df2$m_diff <- abs(df2$m_diff)

# normalize to the same [0, 1] scale; you can choose other values, if appropriate
max_val <- c(1,6,6,6,6,6,6,
             6,1,1.5,1,1,
             1,3,3)
# normalize <-
cols <- colnames(df2)
for(i in 2:dim(df2)[2]){
  df2[, i] <- df2[, i]/max_val[i-1]
} 

ggcols <- c("#F8766D", "#7CAE00", "#00BFC4", "#C77cFF")

# inverted ICC -- higer ICC, close to 1 is better! 
df2$icc.scan <- -df2$icc.scan + 1
df2$icc.diff <- -df2$icc.diff + 1

ggradar(df2, 
        grid.min = 0,
        grid.max = 1,
        values.radar = c("", "", ""),
        group.point.size=1,
        group.colours = ggcols,
        legend.position = "bottom", 
        axis.labels = c("|ME|",
                        'MAE', 'mMAE', 
                        'MAE\nnew site', 'mMAE\nnew site',
                        'MAE new site\nnew preporcessing', 'mMAE new site\nnew preporcessing',
                        "sig y_s", # TeX("${\sigma}(y'_{scan})$"),
                        "mu(d)", # TeX("$\abs{\mu (d)}$"),
                        "sig(d)", # TeX("${\sigma} (d)$"),
                        "ICC s", # TeX("ICC $y'^{(k)}_{scan}$"),
                        "ICC d", # TeX("ICC $d^{(k)}$"),
                        "|MdE|", 
                        "MAdE", "mMAdE"))
